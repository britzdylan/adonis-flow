import Route from '@ioc:Adonis/Core/Route'

Route.group(() => {
  Route.get('/:username/edit', 'ProfilesController.edit').as('profile.edit')
  Route.put('/:username', 'ProfilesController.update').as('profile.update')
  Route.put('/:username/avatar', 'ProfilesController.updateProfileAvatar').as(
    'profile.updateAvatar'
  )
  Route.delete('/:username/avatar', 'ProfilesController.deleteProfileAvatar').as(
    'profile.deleteAvatar'
  )
})
  .prefix('/profile')
  .middleware('auth')
  .namespace('App/Controllers/Http/User')
